\begin{querybox}{Consulta 2 --- Películas dirigidas por `C.\ Nolan'}
\textit{Encuentre los nombres de todas las películas que fueron dirigidas
por \textnormal{`C.~Nolan'}.}

\[
  \pi_{\,pnombre}\!\Big(\;
    \sigma_{\,pdirector\,=\,\text{`C.\ Nolan'}}\!\left(\, peliculas \,\right)
  \;\Big)
\]
\end{querybox}

\vspace{0.4cm}

% =====================================================================
% Consulta 3
% =====================================================================
\begin{querybox}{Consulta 3 --- Películas estrenadas después de 2010}
\textit{Encuentre los nombres de todas las películas que se estrenaron
después del año 2010.}

\[
  \pi_{\,pnombre}\!\Big(\;
    \sigma_{\,a\tilde{n}o\,>\,2010}\!\left(\, peliculas \,\right)
  \;\Big)
\]
\end{querybox}

\vspace{0.4cm}

% =====================================================================
% Consulta 4
% =====================================================================
\begin{querybox}{Consulta 4 --- Películas SciFi con calificación $> 8.5$}
\textit{Encuentre los nombres de todas las películas de la categoría
\textnormal{`SciFi'} que tienen una calificación mayor a 8.5.}

\[
  \pi_{\,pnombre}\!\Big(\;
    \sigma_{\,categoria\,=\,\text{`SciFi'}
      \;\wedge\;
      calificacion\,>\,8.5}\!\left(\, peliculas \,\right)
  \;\Big)
\]
\end{querybox}

\vspace{0.4cm}

% =====================================================================
% Consulta 5
% =====================================================================
\begin{querybox}{Consulta 5 --- IDs de actores en `Interstellar'}
\textit{Encuentre los identificadores (\textnormal{id}) de todos los
actores que actuaron en la película con nombre \textnormal{`Interstellar'}.}

\[
  \pi_{\,id\_actor}\!\Big(\;
    actuo\_en
    \;\bowtie_{id\_pelicula\,=\,id}\;
    \sigma_{\,pnombre\,=\,\text{`Interstellar'}}\!\left(\, peliculas \,\right)
  \;\Big)
\]
\end{querybox}

\vspace{0.4cm}

% =====================================================================
% Consulta 6
% =====================================================================
\begin{querybox}{Consulta 6 --- Actores en películas dirigidas por `C.\ Nolan'}
\textit{Encuentre los nombres de todos los actores que actuaron en alguna
película dirigida por \textnormal{`C.~Nolan'}.}

Definimos primero las películas de C.\ Nolan:
\[
  P_{Nolan} \;=\;
    \sigma_{\,pdirector\,=\,\text{`C.\ Nolan'}}\!\left(\, peliculas \,\right)
\]

Luego, cruzamos con \texttt{actuo\_en} y \texttt{actores}:
\[
  \pi_{\,anombre}\!\Big(\;
    actores
    \;\bowtie_{id\,=\,id\_actor}\;
    actuo\_en
    \;\bowtie_{id\_pelicula\,=\,id}\;
    P_{Nolan}
  \;\Big)
\]
\end{querybox}

\vspace{0.4cm}

% =====================================================================
% Consulta 7
% =====================================================================
\begin{querybox}{Consulta 7 --- Actores que no actuaron en películas de `C.\ Nolan'}
\textit{Encuentre los actores que no actuaron en ninguna película dirigida
por \textnormal{`C.~Nolan'}.}

Definimos los actores que sí actuaron en películas de C.\ Nolan:
\[
  A_{Nolan} \;=\;
    \pi_{\,anombre}\!\Big(\;
      actores
      \;\bowtie_{id\,=\,id\_actor}\;
      actuo\_en
      \;\bowtie_{id\_pelicula\,=\,id}\;
      \sigma_{\,pdirector\,=\,\text{`C.\ Nolan'}}\!(peliculas)
    \;\Big)
\]

Luego, usamos la diferencia:
\[
  \pi_{\,anombre}\!(actores) \;-\; A_{Nolan}
\]
\end{querybox}

\vspace{0.4cm}

% =====================================================================
% Consulta 8
% =====================================================================
\begin{querybox}{Consulta 8 --- Actores solo en películas `SciFi' o `Adventure'}
\textit{Encuentra los actores que solo actuaron en películas de categoría
\textnormal{`SciFi'} o \textnormal{`Adventure'}.}

Definimos los actores que actuaron en alguna película que \textbf{no} es
SciFi ni Adventure:
\[
  A_{otras} \;=\;
    \pi_{\,anombre}\!\Big(\;
      actores
      \;\bowtie_{id\,=\,id\_actor}\;
      actuo\_en
      \;\bowtie_{id\_pelicula\,=\,id}\;
      \sigma_{\,categoria\,\neq\,\text{`SciFi'}
        \;\wedge\;
        categoria\,\neq\,\text{`Adventure'}}\!(peliculas)
    \;\Big)
\]

Definimos los actores que actuaron en al menos una película:
\[
  A_{todos} \;=\;
    \pi_{\,anombre}\!\Big(\;
      actores
      \;\bowtie_{id\,=\,id\_actor}\;
      actuo\_en
    \;\Big)
\]

Luego, los que solo actuaron en SciFi o Adventure:
\[
  A_{todos} \;-\; A_{otras}
\]
\end{querybox}

\vspace{0.4cm}

% =====================================================================
% Consulta 9
% =====================================================================
\begin{querybox}{Consulta 9 --- Actores que han trabajado con más de un director}
\textit{Actores que han trabajado con más de un director.}

Construimos las parejas (actor, director) renombrando para obtener dos
copias con distintos directores:
\[
  AD \;=\;
    \pi_{\,anombre,\,pdirector}\!\Big(\;
      actores
      \;\bowtie_{id\,=\,id\_actor}\;
      actuo\_en
      \;\bowtie_{id\_pelicula\,=\,id}\;
      peliculas
    \;\Big)
\]

Renombramos una copia del director:
\[
  AD' \;=\;
    \rho_{\,pdirector \to pdirector2}\!(AD)
\]

Actores con al menos dos directores distintos:
\[
  \pi_{\,anombre}\!\Big(\;
    \sigma_{\,pdirector\,\neq\,pdirector2}\!\left(\,
      AD \;\bowtie_{anombre\,=\,anombre}\; AD'
    \,\right)
  \;\Big)
\]
\end{querybox}
